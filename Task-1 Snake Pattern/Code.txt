int led[] = {6,7,8,9,10,11,12,13};
int n = 8;

void setup() {
  for(int i=1;i<n;i++)
  pinMode(led[i], OUTPUT);
}

void loop() {
  for(int i=0;i<n;i++){
    digitalWrite(led[i], HIGH);
    delay(200);
    digitalWrite(led[i], LOW);
  }
  for(int i=n-1;i>=0;i--){
    digitalWrite(led[i], HIGH);
    delay(150);
    digitalWrite(led[i], LOW);
  }
}
